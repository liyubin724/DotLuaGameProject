local oop = require("DotLua/OOP/oop")
local EntityController = require("DotLua/Framework/Entity/EntityController")

local EntityRootController =
    oop.class(
    "EntityRootController",
    function(self)
        self.rootGameObject = nil
        self.rootTransform = nil
    end,
    EntityController
)

function EntityRootController:GetRootGameObject()
    return self.rootGameObject
end

function EntityRootController:SetRootGameObject(rootGO)
    self.rootGameObject = rootGO
    self.rootTransform = rootGO.transform
end

function EntityRootController:GetRootTransform()
    return self.rootTransform
end

function EntityRootController:SetRootTransform(transform)
    self.rootTransform = transform
    self.rootGameObject = transform.gameobject
end

return EntityRootController
