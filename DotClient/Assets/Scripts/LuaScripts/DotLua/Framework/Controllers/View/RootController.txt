local oop = require("DotLua/OOP/oop")
local ActorController = require("DotLua/Framework/ActorController")

local RootController =
    oop.class(
    "RootController",
    function(self)
        self.rootGameObject = nil
        self.rootTransform = nil
    end,
    ActorController
)

function RootController:GetRootGameObject()
    return self.rootGameObject
end

function RootController:SetRootGameObject(rootGO)
    self.rootGameObject = rootGO
    self.rootTransform = rootGO.transform
end

function RootController:GetRootTransform()
    return self.rootTransform
end

function RootController:SetRootTransform(transform)
    self.rootTransform = transform
    self.rootGameObject = transform.gameobject
end

return RootController
