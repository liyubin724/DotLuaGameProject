local oop = require('DotLua/ECS/OOP/oop')
local System = require('DotLua/ECS/Systems/System')

local FSMStateSystem =
    oop.class(
    'DotLua.ECS.Systems.FSM.FSMStateSystem',
    function(self,stateName)
        self.stateName = stateName
    end,
    System
)

function FSMStateSystem:GetStateName()
    return self.stateName
end

function FSMStateSystem:GetNextStateName(entity)
    return self.stateName
end

function FSMStateSystem:DoEnter(entity,fromStateName)
end

function FSMStateSystem:DoExit(entity,toStateName)
end

function FSMStateSystem:DoExecute(entity)
end

return FSMStateSystem
