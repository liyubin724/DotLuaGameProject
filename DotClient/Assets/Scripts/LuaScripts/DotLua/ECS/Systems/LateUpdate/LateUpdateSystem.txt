local oop = require("DotLua/OOP/oop")
local System = oop.using("DotLua/ECS/Systems/System")
local Const = require("DotLua/ECS/Const")

local LateUpdateSystem =
    oop.class(
    "DotLua.ECS.Systems.LateUpdate.LateUpdateSystem",
    function(self,frequencyInTime)
        self.frequencyInTime = frequencyInTime or Const.TIME_EVERY_UPDATE
        self.elapseDeltaTime = 0
    end,
    System
)

function LateUpdateSystem:GetFrequency()
    return self.frequencyInTime
end

function LateUpdateSystem:DoLateUpdate(deltaTime, unscaleDeltaTime)
    self.elapseDeltaTime = self.elapseDeltaTime + deltaTime

    if self.elapseDeltaTime >= self.frequencyInTime then
        if self.frequencyInTime <= 0 then
            self:onLateUpdate(self.elapseDeltaTime)
            self.elapseDeltaTime = 0
        else
            self:onTimeUpdate(self.frequencyInTime)
            self.elapseDeltaTime = self.elapseDeltaTime - self.frequencyInTime
        end
    end
end

function LateUpdateSystem:onLateUpdate(deltaTime)
end

return LateUpdateSystem
