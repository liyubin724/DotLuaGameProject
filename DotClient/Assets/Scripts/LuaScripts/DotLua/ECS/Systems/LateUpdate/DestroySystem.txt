local oop = require("DotLua/OOP/oop")
local LateUpdateSystem = require("DotLua/ECS/Systems/LateUpdate/LateUpdateSystem")
local GroupEvent = require("DotLua/ECS/Core/GroupEvent")
local Collector = require("DotLua/ECS/Core/Collector")

local DestroySystem =
    oop.class(
    "DotLua.ECS.Systems.LateUpdate.LateUpdateSystem",
    function(self, frequencyInTime, group)
        self.collector = Collector(group,GroupEvent.Added)
    end,
    LateUpdateSystem
)

function DestroySystem:onInitlize()
    self.collector:DoActivate()
end

function DestroySystem:onDestroy()
    self.collector:DoDeactivate()
end

function DestroySystem:onLateUpdate(deltaTime)
    
end

return DestroySystem
