local oop = require("DotLua/OOP/oop")
local Component = require("DotLua/ECS/Core/Component")

local ParentComponent =
    oop.class(
    "DotLua.ECS.Components.Basic.Dependencies.ParentComponent",
    function(self)
        self.guid = nil
    end,
    Component
)

function ParentComponent:GetParent()
    return self.guid
end

function ParentComponent:SetParent(guid)
    self.guid = guid
end

function ParentComponent:DoGet(guid)
    self.guid = guid
end

function ParentComponent:DoRelease()
    self.guid = nil
end

return ParentComponent
