local oop = require("DotLua/OOP/oop")
local Component = require("DotLua/ECS/Core/Component")

local GUIDComponent =
    oop.class(
    "DotLua.ECS.Components.Basic.Identifies.GUIDComponent",
    function(self)
        self.guid = nil
    end,
    Component
)

function GUIDComponent:GetGUID()
    return self.guid
end

function GUIDComponent:DoGet(guid)
    self.guid = guid
end

function GUIDComponent:DoRelease()
    self.guid = nil
end

return GUIDComponent
