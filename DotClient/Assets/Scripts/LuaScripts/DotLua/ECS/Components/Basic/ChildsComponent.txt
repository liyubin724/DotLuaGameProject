local oop = require("DotLua/OOP/oop")
local Component = require("DotLua/ECS/Components/Component")
local ListDictionary = require("DotLua/Generic/Collections/ListDictionary")

local ChildsComponent =
    oop.class(
    "DotLua.ECS.Components.Basic.ChildsComponent",
    function(self)
        self.childListDic = ListDictionary()
    end,
    Component
)

function ChildsComponent:GetChilds()
    return self.childListDic:GetItems()
end

function ChildsComponent:AddEntity(guid)
    if not self.childListDic:HasItem(guid) then
        self.childListDic:AddItem(guid,guid)
    end
end

function ChildsComponent:RemoveEntity(guid)
    self.childListDic:RemoveItem(guid)
end

function ChildsComponent:HasEntity(guid)
    return self.childListDic:HasItem(guid)
end

function ChildsComponent:DoRelease()
    self.childListDic:Clear()
end

return ChildsComponent
