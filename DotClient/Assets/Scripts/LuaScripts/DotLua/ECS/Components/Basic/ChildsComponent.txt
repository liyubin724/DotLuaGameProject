local oop = require("DotLua/OOP/oop")
local Component = require("DotLua/ECS/Components/Component")

local tclear = table.clear
local tinsert = table.insert
local tremovevalue = table.removevalue
local tcontainsvalue = table.containsvalue

local ChildsComponent =
    oop.class(
    "DotLua.ECS.Components.Basic.ChildsComponent",
    function(self)
        self.childs = {}
    end,
    Component
)

function ChildsComponent:GetChilds()
    return self.childs
end

function ChildsComponent:Add(guid)
    if not tcontainsvalue(self.childs, guid) then
        tinsert(self.childs, guid)
    end
end

function ChildsComponent:Remove(guid)
    if tcontainsvalue(self.childs, guid) then
        tremovevalue(self.childs, guid)
        return true
    end

    return false
end

function ChildsComponent:Has(guid)
    return tcontainsvalue(self.childs, guid)
end

function ChildsComponent:DoRelease()
    tclear(self.childs)
end

return ChildsComponent
