local oop = require("DotLua/OOP/oop")

local TimerRunningComponent =
    oop.class(
    "DotLua.ECS.Components.Timer.TimerRunningComponent",
    function(self)
        self.timerIndex = -1
    end
)

function TimerRunningComponent:GetTimerIndex()
    return self.timerIndex
end

function TimerRunningComponent:DoGet(timerIndex)
    self.timerIndex = timerIndex
end

function TimerRunningComponent:DoRelease()
    self.timerIndex = -1
end

return TimerRunningComponent
