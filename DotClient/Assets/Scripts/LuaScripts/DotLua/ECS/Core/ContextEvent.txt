local oop = require('DotLua/OOP/oop')

local ContextEvent =
    oop.enum(
    'DotLua.ECS.Contexts.ContextEvent',
    {
        EntityCreated = 1,
        EntityReleased = 2,
        EntityChangedForAddComponent = 3,
        EntityChangedForRemoveComponent = 4,
        EntityChangedForReplaceComponent = 5,
        EntityChangedForModifyComponent = 6,
    }
)

return ContextEvent
