local oop = require("DotLua/OOP/oop")
local ObjectPool = require("DotLua/Pool/ObjectPool")
local EntityIdentity = require("DotLua/ECS/Core/EntityIdentity")

local pool = ObjectPool(EntityIdentity)
local LogTag = "EntityIdentityPool"

local EntityIdentityPool = {}
function EntityIdentityPool.Get(contextName, guid)
    return pool:Get(contextName, guid)
end

function EntityIdentityPool.Release(entityIdentity)
    if oop.isDebug then
        if not entityIdentity or not oop.isinstanceof(entityIdentity, EntityIdentity) then
            oop.error(LogTag, "The entityIdentity is nil or is not a instance of EntityIdentity")
            return
        end
    end

    pool:Release(entityIdentity)
end

return EntityIdentityPool
