local oop = require("DotLua/OOP/oop")

local tremove = table.remove
local tinsert = table.insert
local tclear = table.clear

local getFuncName = "DoGet"
local releaseFuncName = "DoRelease"

local ObjectPool =
    oop.class(
    "DotLua.Pool.ObjectPool",
    function(self, itemClass)
        self.itemClass = itemClass
        self.unusedItems = {}
    end
)

function ObjectPool:Get(...)
    local item

    if #(self.unusedItems) > 0 then
        item = self.unusedItems[1]
        tremove(self.unusedItems, 1)
    else
        item = self.itemClass()
    end

    local getFunc = item[getFuncName]
    if getFunc then
        getFunc(item, ...)
    end

    return item
end

function ObjectPool:Release(item, ...)
    if item then
        local releaseFunc = item[releaseFuncName]
        if releaseFunc then
            releaseFunc(item, ...)
        end

        tinsert(self.unusedItems, item)
    end
end

function ObjectPool:Clear()
    tclear(self.unusedItems)
end

return ObjectPool
