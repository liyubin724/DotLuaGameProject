local oop = require('DotLua/OOP/oop')

local Component = require('DotLua/ECS/Core/Component')

local SkeletionLogicComponent =
    oop.class(
    'Game.ECS.Components.Avatar.SkeletionLogicComponent',
    function(self)
        self.skeletionPath = nil
    end,
    Component
)

function SkeletionLogicComponent:GetSkeletionPath()
    return self.skeletionPath
end

function SkeletionLogicComponent:SetSkeletionPath(skeletionPath)
    self.skeletionPath = skeletionPath
end

return SkeletionLogicComponent
