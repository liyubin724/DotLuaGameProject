local oop = require('DotLua/ECS/OOP/oop')
local GUID = require("DotLua/Generic/GUID")

local FSMToken =
    oop.class(
    'DotLua.ECS.Systems.FSM.Token.FSMToken',
    function(self, name)
        self.name = name
        self.guid = GUID.Create()
    end
)

function FSMToken:GetName()
    return self.name
end

FSMToken.__eq = function(t1,t2)
    return t1.guid == t2.guid
end

FSMToken.__tostring = function(token)
    return "Token("..token.name..":"..tostring(token.guid)..")"
end

return FSMToken
