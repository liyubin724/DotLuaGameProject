local oop = require('DotLua/ECS/OOP/oop')
local FSMEventArgs = require('DotLua/ECS/Systems/FSM/EventArgs/FSMEventArgs')

local FSMEnterStateEventArgs =
    oop.class(
    'DotLua.ECS.Systems.FSM.EventArgs.FSMEnterStateEventArgs',
    function(self, fromToken, userdata)
        self.fromToken = fromToken
        self.userdata = userdata
    end,
    FSMEventArgs
)

function FSMEnterStateEventArgs:GetFromToken()
    return self.fromToken
end

return FSMEnterStateEventArgs
