local oop = require('DotLua/ECS/OOP/oop')
local FSMEventArgs = require('DotLua/ECS/Systems/FSM/EventArgs/FSMEventArgs')

local FSMExitStateEventArgs =
    oop.class(
    'DotLua.ECS.Systems.FSM.EventArgs.FSMExitStateEventArgs',
    function(self, toToken, userdata)
        self.toToken = toToken
        self.userdata = userdata
    end,
    FSMEventArgs
)

function FSMExitStateEventArgs:GetToToken()
    return self.toToken
end

return FSMExitStateEventArgs
