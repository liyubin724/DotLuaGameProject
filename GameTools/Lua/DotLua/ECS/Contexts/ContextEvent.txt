local oop = require('DotLua/OOP/oop')

local ContextEvent =
    oop.enum(
    'DotLua.ECS.Contexts.ContextEvent',
    {
        EntityCreated = 1,
        EntityReleased = 2,
        ComponentAdded = 1,
        ComponentRemoved = 2,
        ComponentReplaced = 3,
        ComponentModified = 4,
        GroupCreated = 5
    }
)

return ContextEvent
