local oop = require('DotLua/OOP/oop')

local ContextEvent =
    oop.enum(
    'DotLua.ECS.Contexts.ContextEvent',
    {
        EntityCreated = 1,
        EntityReleased = 2,
        ComponentAdded = 3,
        ComponentRemoved = 4,
        ComponentReplaced = 5,
        ComponentModified = 6,
        GroupCreated = 7,
        GroupReleased = 8
    }
)

return ContextEvent
