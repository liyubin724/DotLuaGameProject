local oop = require('DotLua/OOP/oop')
local Component = require('DotLua/ECS/Components/Component')

local PrevStateComponent =
    oop.class(
    'DotLua.ECS.Components.FSM.PrevStateComponent',
    function(self)
        self.stateName = nil
        self.userdata = nil
    end,
    Component
)

function PrevStateComponent:GetStateName()
    return self.stateName
end

function PrevStateComponent:GetUserdata()
    return self.userdata
end

function PrevStateComponent:DoGet(name, stateName, userdata)
    self:GetBaseClass().DoGet(self, name)

    self.stateName = stateName
    self.userdata = userdata
end

function PrevStateComponent:DoRelease()
    self:GetBaseClass().DoRelease(self)

    self.stateName = nil
    self.userdata = nil
end

return PrevStateComponent
