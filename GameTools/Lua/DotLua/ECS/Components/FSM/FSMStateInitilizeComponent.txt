local oop = require('DotLua/OOP/oop')
local Component = require('DotLua/ECS/Components/Component')

local FSMStateInitilizeComponent =
    oop.class(
    'DotLua.ECS.Components.FSM.FSMStateInitilizeComponent',
    function(self)
        self.initStateName = nil
    end,
    Component
)

function FSMStateInitilizeComponent:GetInitStateName()
    return self.initStateName
end

function FSMStateInitilizeComponent:SetInitStateName(initStateName)
    self.initStateName = initStateName
end

function FSMStateInitilizeComponent:DoGet(initStateName)
    self.initStateName = initStateName
end

function FSMStateInitilizeComponent:DoRelease()
    self.initStateName = nil
end

return FSMStateInitilizeComponent
