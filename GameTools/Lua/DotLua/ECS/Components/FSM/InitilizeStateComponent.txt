local oop = require('DotLua/OOP/oop')
local Component = require('DotLua/ECS/Components/Component')

local InitilizeStateComponent =
    oop.class(
    'DotLua.ECS.Components.FSM.InitilizeStateComponent',
    function(self)
        self.stateName = nil
        self.userdata = nil
    end,
    Component
)

function InitilizeStateComponent:GetStateName()
    return self.stateName
end

function InitilizeStateComponent:DoGet(name, stateName,userdata)
    self:GetBaseClass().DoGet(self, name)

    self.stateName = stateName
    self.userdata = userdata
end

function InitilizeStateComponent:DoRelease()
    self:GetBaseClass().DoRelease(self)

    self.stateName = nil
    self.userdata = nil
end

return InitilizeStateComponent
