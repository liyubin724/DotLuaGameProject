local oop = require('DotLua/OOP/oop')
local Component = require('DotLua/ECS/Components/Component')

local FSMStateComponent =
    oop.class(
    'DotLua.Components.FSM.FSMStateComponent',
    function(self)
        self.stateName = nil
        self.isRunning = true
    end,
    Component
)

function FSMStateComponent:GetIsRuning()
    return self.isRunning
end

function FSMStateComponent:SetIsRunning(isRunning)
    self.isRunning = isRunning
end

function FSMStateComponent:GetStateName()
    return self.stateName
end

function FSMStateComponent:SetStateName(stateName)
    self.stateName = stateName
end

function FSMStateComponent:DoGet()
    self.isRuning = true
end

function FSMStateComponent:DoRelease()
    self.stateName = nil
    self.isRunning = false
end

return FSMStateComponent
