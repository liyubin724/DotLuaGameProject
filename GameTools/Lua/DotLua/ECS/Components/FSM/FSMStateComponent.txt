local oop = require('DotLua/OOP/oop')
local Component = require('DotLua/ECS/Components/Component')

local FSMStateComponent =
    oop.class(
    'DotLua.ECS.Components.FSM.FSMStateComponent',
    function(self)
        self.prevStateName = nil
        self.currentStateName = nil
        self.nextStateName = nil
    end,
    Component
)

function FSMStateComponent:GetPrevStateName()
    return self.prevStateName
end

function FSMStateComponent:SetPrevStateName(stateName)
    self.prevStateName = stateName
end

function FSMStateComponent:GetCurrentStateName()
    return self.currentStateName
end

function FSMStateComponent:SetCurrentStateName(stateName)
    self.currentStateName = stateName
end

function FSMStateComponent:GetNextStateName()
    return self.nextStateName
end

function FSMStateComponent:SetNextStateName(stateName)
    self.nextStateName = stateName
end

function FSMStateComponent:DoRelease()
    self.prevStateName = nil
    self.currentStateName = nil
    self.nextStateName = nil
end

return FSMStateComponent
