local oop = require('DotLua/OOP/oop')
local Component = require('DotLua/ECS/Components/Component')

local NextStateComponent =
    oop.class(
    'DotLua.ECS.Components.FSM.NextStateComponent',
    function(self)
        self.stateName = nil
        self.userdata = nil
    end,
    Component
)

function NextStateComponent:GetStateName()
    return self.stateName
end

function NextStateComponent:GetUserdata()
    return self.userdata
end

function NextStateComponent:DoGet(name, stateName, userdata)
    self:GetBaseClass().DoGet(self, name)

    self.stateName = stateName
    self.userdata = userdata
end

function NextStateComponent:DoRelease()
    self:GetBaseClass().DoRelease(self)

    self.stateName = nil
    self.userdata = nil
end

return NextStateComponent
