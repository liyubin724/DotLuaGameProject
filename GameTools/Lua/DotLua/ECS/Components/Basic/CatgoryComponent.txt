local oop = require('DotLua/OOP/oop')
local Component = require('DotLua/Components/Component')

local CatgoryComponent =
    oop.class(
    'DotLua.Components.Basic.CatgoryComponent',
    function(self)
        self.catgory = nil
    end,
    Component
)

function CatgoryComponent:GetCatgory()
    return self.catgory
end

function CatgoryComponent:SetCatgory(catgory)
    self.catgory = catgory
end

function CatgoryComponent:Reset()
    self.catgory = nil
end

return CatgoryComponent
