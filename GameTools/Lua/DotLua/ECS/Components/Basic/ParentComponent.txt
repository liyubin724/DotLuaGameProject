local oop = require('DotLua/OOP/oop')
local Component = require('DotLua/ECS/Components/Component')

local ParentComponent =
    oop.class(
    'DotLua.ECS.Components.ParentComponent',
    function(self)
        self.parentGUID = nil
    end,
    Component
)

function ParentComponent:GetParentGUID()
    return self.parentGUID
end

function ParentComponent:SetParentGUID(parentID)
    self.parentGUID = parentID
end

function ParentComponent:DoGet(parentGUID)
    self.parentGUID = parentGUID
end

function ParentComponent:DoRelease()
    self.parentGUID = nil
end

return ParentComponent
