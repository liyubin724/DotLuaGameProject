local oop = require("DotLua/OOP/oop")
local ecs = require('DotLua/ECS/ecs')
local init = require("DotLua/init")

GameLauncher = {}
GameLauncher.__index = GameLauncher

function GameLauncher.DoStart()
    _G.isDebug = true
    
    _G.GameApp = require("Game/GameApplication")

end

function GameLauncher.DoUpdate(deltaTime, unscaleDeltaTime)
    ecs.update(deltaTime, unscaleDeltaTime)
end

function GameLauncher.DoLateUpdate()
    ecs.lateupdate()
end

function GameLauncher.DoDestroy()
    ecs.teardown()
end
