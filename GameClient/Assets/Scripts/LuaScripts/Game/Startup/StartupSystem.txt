local oop = require('DotLua/OOP/oop')
local UIRootComponent = require "Game.UI.Root.UIRootComponent"
local InitializeSystem = oop.using('DotLua/ECS/Systems/LifeCycle/InitializeSystem')

local DEUIRoot = CS.DotEngine.UI.UIRoot

local StartupSystem =
    oop.class(
    'StartupSystem',
    function(self)
    end,
    InitializeSystem
)

function StartupSystem:DoInitialize()
    local gameECS = _G.gameECS
    local uiRootEntity = gameECS:GetUIRootEntity()
    uiRootEntity:AddComponent(DEUIRoot.Root:GetRootComponent())
end

return StartupSystem
