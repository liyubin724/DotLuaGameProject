local oop = require('DotLua/OOP/oop')
local Context = oop.using('DotLua/ECS/Contexts/Context')

local UIContext =
    oop.class(
    'UIContext',
    function(self, name)
        self.uiRootEntity = nil
    end,
    Context
)

function UIContext:SetRootEntity(entity)
    self.uiRootEntity = entity
end

function UIContext:GetRootEntity()
    return self.uiRootEntity
end

return UIContext
