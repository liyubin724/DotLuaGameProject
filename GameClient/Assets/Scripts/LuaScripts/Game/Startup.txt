Game = {}
Game.IsDebug = true

require('DotLua/Utility/math')
require('DotLua/Utility/table')
require('DotLua/Class')
require('DotLua/Log/DebugLog')
require('DotLua/Config/ConfigManager')
require('DotLua/Asset/AssetManager')
require('DotLua/Event/EventDispatcher')
require('DotLua/Timer/HierarchicalTimerWheel')
require('DotLua/UI/UIPanelManager')
require("DotLua/GOPool/GameObjectPoolManager")
require('Game/EnvManager')

if Game.IsDebug then
    require('Debugger/LuaPanda').start('127.0.0.1', 8818)

    DebugLog.SetEnable(true, true, true)
else
    DebugLog.SetEnable(true, false, false)
end

local function InitMgr()
    Game.ConfigMgr = ConfigManager("Game/Config/")
    Game.TimerMgr = HierarchicalTimerWheel()
    Game.EventMgr = EventDispatcher()
    Game.AssetMgr = AssetManager()
    Game.PanelMgr = UIPanelManager()
    Game.PoolMgr = GameObjectPoolManager()

    Game.EnvMgr = EnvManager()
end

InitMgr()
