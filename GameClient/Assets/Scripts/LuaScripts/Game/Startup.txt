Game = Game or {}
Game.IsDebug = true

require('DotLua/Utility/math')
require('DotLua/Utility/table')
require('DotLua/OOP/OOP')
require('DotLua/Log/DebugLog')

if Game.IsDebug then
    require('Debugger/LuaPanda').start('127.0.0.1', 8818)

    DebugLog.SetEnable(true, true, true)
else
    DebugLog.SetEnable(true, false, false)
end

local function InitMgr()
    Game.ConfigMgr = instanceWithParams('DotLua/Config/ConfigManager', 'Game/Config/')
    Game.TimerMgr = instance('DotLua/Timer/HierarchicalTimerWheel')
    Game.EventMgr = instance('DotLua/Event/EventDispatcher')
    Game.AssetMgr = instance('DotLua/Asset/AssetManager')
    Game.PanelMgr = instance('DotLua/UI/UIPanelManager')
    Game.PoolMgr = instance('DotLua/GOPool/GameObjectPoolManager')
    Game.EnvMgr = instance('Game/EnvManager')
end

InitMgr()
