local ecs = require("DotLua/ECS/ecs")

local oop = require("DotLua/OOP/oop")
oop.info("Test","Test Timer")

local HTimer = ecs.getservicer("HTimer")

TestTimer = {}
TestTimer.__index = TestTimer

function TestTimer:HandIntervalTimer(userdata)
    oop.info("Test","TestTimer:HandIntervalTimer->userdata = "..userdata)
end

function TestTimer:HandTickTimer(userdata)
    oop.info("Test","TestTimer:HandTickTimer->userdata = "..userdata)
end

local index = HTimer:AddIntervalTimer(0.5,TestTimer,TestTimer.HandIntervalTimer,"Test11111111111")
local index2 = HTimer:AddIntervalTimer(0.6,TestTimer,TestTimer.HandIntervalTimer,"Test22222222222")
--HTimer:AddTickTimer(TestTimer,TestTimer.HandTickTimer,"Test Tick Timer")

oop.info("Test","TimerIndex = "..tostring(index))
