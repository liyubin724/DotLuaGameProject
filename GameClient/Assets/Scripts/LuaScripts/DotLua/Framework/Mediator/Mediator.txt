local Notifier = require('DotLua/Framework/Observer/Notifier')

local Mediator =
    class(
    Notifier,
    function(self, name)
        self.name = name
        self.view = nil
    end
)

function Mediator:GetName()
    return self.name
end

function Mediator:IsViewValid()
    return self.view ~= nil
end

function Mediator:GetView()
    return self.view
end

function Mediator:SetView(view)
    self.view = view
end

function Mediator:ListNotificationInterests()
    return {}
end

function Mediator:HandleNotification(notification)
end

function Mediator:DoRegister()
end

function Mediator:DoRemove()
end

return Mediator
