local Notification =
    class(
    function(self, name, body, userdata)
        self.name = name
        self.body = body
        self.userdata = userdata
    end
)

function Notification:GetName()
    return self.name
end

function Notification:GetBody()
    return self.body
end

function Notification:GetUserdata()
    return self.userdata
end

return Notification
