local Notifier =
    class(
    function(self)
        self.uniqueID = nil
    end
)

function Notifier:GetUniqueID()
    return self.uniqueID
end

function Notifier:OnInitialized(uniqueID)
    self.uniqueID = uniqueID
end

function Notifier:SendNotification(name, body, userdata)
    local Framework = require('DotLua/Framework/Framework')
    local facade = Framework.GetFacade(self.uniqueID)
    if facade then
        facade:SendNotification(name, body, userdata)
    else
        error('')
    end
end

return Notifier
