local oop = require('DotLua/OOP/oop')

local Component = oop.using('DotLua/ECS/Components/Component')

local ViewComponent =
    oop.class(
    'ViewComponent',
    function(self)
        self.entity = nil
    end,
    Component
)

function ViewComponent:SetEntity(entity)
    self.entity = entity
end

return ViewComponent
