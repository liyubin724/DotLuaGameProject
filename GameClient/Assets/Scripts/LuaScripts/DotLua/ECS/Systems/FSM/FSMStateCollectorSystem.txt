local oop = require('DotLua/ECS/OOP/oop')
local CollectorUpdateSystem = require('DotLua/ECS/Systems/Collectors/CollectorUpdateSystem')

local FSMStateCollectorSystem =
    oop.class(
    'DotLua.ECS.Systems.FSM.FSMStateCollectorSystem',
    function(self, context)
        self.matchine = nil
    end,
    CollectorUpdateSystem
)

function FSMStateCollectorSystem:SetMatchine(matchine)
    self.matchine = matchine
end

return FSMStateCollectorSystem
