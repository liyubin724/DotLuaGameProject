local UEApplication = CS.UnityEngine.Application

local App = {}

App.platform = nil
App.persistentDataPath = nil
App.systemLanguage = nil

function App.GetPlatform()
    if not App.platform then
        App.platform = UEApplication.platform
    end
    return App.platform
end

function App.GetPersistentDataPath()
    if not App.persistentDataPath then
        App.persistentDataPath = UEApplication.persistentDataPath
    end
    return App.persistentDataPath
end

function App.GetSystemLanguage()
    if not App.systemLanguage then
        App.systemLanguage = CS.UnityEngine.SystemLanguage(UEApplication.systemLanguage)
    end

    return App.systemLanguage
end

return App
